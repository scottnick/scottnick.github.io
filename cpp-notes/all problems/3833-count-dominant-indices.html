class Solution {
public:
    int dominantIndices(vector<int>& nums) {
        if (nums.size() == 1) {
            return 0;
        }
        int n = nums.size();
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        int count = 0;
        for (int j = 0; j < n - 1; j++) {
            sum -= nums[j];
            if ((n - j - 1) * nums[j] > sum) {
                count++;
            }
        }
        return count;
    }
};
