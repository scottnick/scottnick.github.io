class Solution {
public:
    int minimumPairRemoval(vector<int>& nums) {
        bool check = true;
        int time = 0;
        while (check) {
            int count = 2147483647;
            int pos = 0;
            if (nums.size() < 2) {
                return time;
            }
            check = false;
            for (int i = 0; i < nums.size() - 1; i++) {
                if (nums[i] > nums[i + 1]) {
                    check = true;
                }
                if (nums[i] + nums[i + 1] < count) {
                    pos = i;
                    count = nums[i] + nums[i + 1];
                }
            }
            if (check) {
                nums[pos] = nums[pos] + nums[pos + 1];
                nums.erase(nums.begin() + pos + 1);
                time++;
            }
        }
        return time;
    }
};
