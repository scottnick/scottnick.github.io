class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        vector<int> nums;
        int pos1 = 0, pos2 = 0;
        while (pos1 < m && pos2 < n) {
            if (nums1[pos1] < nums2[pos2]) {
                nums.push_back(nums1[pos1]);
                pos1++;
            } else {
                nums.push_back(nums2[pos2]);
                pos2++;
            }
        }
        while (pos1 < m) {
            nums.push_back(nums1[pos1]);
            pos1++;
        }
        while (pos2 < n) {
            nums.push_back(nums2[pos2]);
            pos2++;
        }
        nums1.resize(m + n);
        for (int i = 0; i < m + n; i++) {
            nums1[i] = nums[i];
        }
    }
};
