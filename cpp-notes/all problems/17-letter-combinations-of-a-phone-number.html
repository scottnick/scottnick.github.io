class Solution {
public:
    vector<string> letterCombinations(string digits) {
        vector<int> digit;
        int n = digits.size();
        for (int i = 0; i < n; i++) {
            digit.push_back((int)digits[i] - (int)'0');
        }
        vector<string> letter;
        vector<vector<char>> eng = {{},
                                    {},
                                    {'a', 'b', 'c'},
                                    {'d', 'e', 'f'},
                                    {'g', 'h', 'i'},
                                    {'j', 'k', 'l'},
                                    {'m', 'n', 'o'},
                                    {'p', 'q', 'r', 's'},
                                    {'t', 'u', 'v'},
                                    {'w', 'x', 'y', 'z'}};
        vector<int> comb(n, 0);
        while (comb[0] < eng[digit[0]].size()) {
            string s = "";
            for (int j = 0; j < n; j++) {
                s = s + eng[digit[j]][comb[j]];
            }
            letter.push_back(s);
            comb[n - 1]++;
            for (int k = n - 1; k > 0; k--) {
                if (comb[k] >= eng[digit[k]].size()) {
                    comb[k] = 0;
                    comb[k - 1]++;
                }
            }
        }
        return letter;
    }
};
