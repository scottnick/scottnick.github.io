class Solution {
public:
    bool isDigitorialPermutation(int n) {
        int fact[10] = {1, 1, 2, 6, 24, 120, 720, 5040, 40320, 362880};
        int sum = 0;
        int k = n;
        while (k > 0) {
            sum += fact[k % 10];
            k /= 10;
        }
        int a[10] = {0}, b[10] = {0};
        while (n > 0) {
            a[n % 10]++;
            n /= 10;
        }
        while (sum > 0) {
            b[sum % 10]++;
            sum /= 10;
        }
        for (int i = 0; i < 10; i++) {
            if (a[i] != b[i]) {
                return false;
            }
        }
        return true;
    }
};
