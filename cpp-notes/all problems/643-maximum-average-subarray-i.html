class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        int sum = 0;
        int nsum = 0;
        double x;
        for (int i = 0; i < k; i++) {
            sum += nums[i];
        }
        nsum = sum;
        for (int j = k; j < nums.size(); j++) {
            nsum = nsum + nums[j] - nums[j - k];
            sum = max(sum, nsum);
        }
        x = (double)sum / k;
        return x;
    }
};
