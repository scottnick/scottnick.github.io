class Solution {
public:
    char nextGreatestLetter(vector<char>& letters, char target) {
        int n = letters.size();
        if (letters[n - 1] <= target) {
            return letters[0];
        }
        for (char a : letters) {
            if (a > target) {
                return a;
            }
        }
        return letters[0];
    }
};
