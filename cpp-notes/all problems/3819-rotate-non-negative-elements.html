class Solution {
public:
    vector<int> rotateElements(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> spos;
        vector<int> snum;
        for (int i = 0; i < n; i++) {
            if (nums[i] >= 0) {
                spos.push_back(i);
                snum.push_back(nums[i]);
            }
        }
        int m = spos.size();
        for (int j = 0; j < m; j++) {
            nums[spos[j]] = snum[(j + k) % m];
        }
        return nums;
    }
};
