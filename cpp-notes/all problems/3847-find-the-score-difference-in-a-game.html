class Solution {
public:
    int scoreDifference(vector<int>& nums) {
        int n = nums.size();
        bool act = true;
        int first = 0, second = 0;
        for (int i = 0; i < n; i++) {
            if (nums[i] % 2 == 1) {
                act = !act;
            }
            if ((i + 1) % 6 == 0) {
                act = !act;
            }
            if (act) {
                first += nums[i];
            } else {
                second += nums[i];
            }
        }
        return first - second;
    }
};
