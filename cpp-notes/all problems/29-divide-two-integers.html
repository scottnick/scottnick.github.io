class Solution {
public:
    int divide(int dividend, int divisor) {
        if (dividend == INT_MIN && divisor == -1) {
            return INT_MAX;
        }
        bool sign = (dividend > 0) ^ (divisor > 0); // XOR, trueç‚ºè² 
        long long a = llabs((long long)dividend);
        long long b = llabs((long long)divisor);
        long long ans = 0;
        while (a >= b) {
            long long temp = b;
            long long mul = 1;
            while (temp + temp <= a) {
                temp += temp;
                mul += mul;
            }
            a -= temp;
            ans += mul;
        }
        return (sign ? -ans : ans);
    }
};
