class Solution {
public:
    int minimumPrefixLength(vector<int>& nums) {
        int last = nums[0];
        int count = 0;
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] <= last) {
                count = i;
            }
            last = nums[i];
        }
        return count;
    }
};
