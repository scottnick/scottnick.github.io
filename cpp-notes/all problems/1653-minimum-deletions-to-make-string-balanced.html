class Solution {
public:
    int minimumDeletions(string s) {
        int b = 0;
        int ans = 0;
        for (char r : s) {
            if (r == 'b') {
                b++;
            } else {
                ans = min(ans + 1, b);
            }
        }
        return ans;
    }
};
