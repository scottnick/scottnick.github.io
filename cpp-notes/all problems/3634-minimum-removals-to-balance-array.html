class Solution {
public:
    int minRemoval(vector<int>& nums, int k) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        int r = 0;
        int best = 1;

        for (int l = 0; l < n; l++) {
            if (r < l) {
                r = l;
            }
            while (r < n && (long long)nums[r] <= (long long)nums[l] * k) {
                r++;
            }
            best = max(best, r - l);
        }

        return n - best;
    }
};
