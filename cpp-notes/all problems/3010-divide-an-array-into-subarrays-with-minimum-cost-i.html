class Solution {
public:
    int minimumCost(vector<int>& nums) {
        int fst = 70, snd = 100;
        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] < fst) {
                snd = fst;
                fst = nums[i];
            } else if (nums[i] < snd) {
                snd = nums[i];
            }
        }
        return nums[0] + fst + snd;
    }
};
