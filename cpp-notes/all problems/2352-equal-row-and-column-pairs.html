class Solution {
public:
    int equalPairs(vector<vector<int>>& grid) {
        int count = 0;
        for (int i = 0; i < grid.size(); i++) {
            bool check = true;
            for (int j = 0; j < grid.size(); j++) {
                check = true;
                for (int k = 0; k < grid.size(); k++) {
                    if (grid[i][k] != grid[k][j]) {
                        check = false;
                        break;
                    }
                }
                if (check) {
                    count++;
                }
            }
        }
        return count;
    }
};
