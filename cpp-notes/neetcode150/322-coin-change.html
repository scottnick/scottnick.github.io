class Solution {
public:
    vector<int> better;
    vector<int> coin;

    void charge(int now, int tol, int amo) {
        if (now <= amo) {
            if (tol < better[now]) {
                better[now] = tol;
                for (int i = 0; i < coin.size(); i++) {
                    charge(now + coin[i], tol + 1, amo);
                }
            }
        }
    }

    int coinChange(vector<int>& coins, int amount) {
        better.assign(amount + 1, 20000);
        for (int a : coins) {
            if (a <= amount) {
                coin.push_back(a);
            }
        }
        charge(0, 0, amount);
        return (better[amount] == 20000 ? -1 : better[amount]);
    }
};
