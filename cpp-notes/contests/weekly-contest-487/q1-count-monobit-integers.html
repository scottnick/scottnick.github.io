class Solution {
public:
    int countMonobit(int n) {
        vector<bool> mon(1001, false);
        mon[0] = true;
        mon[1] = true;
        int count = 0;
        for (int i = 3; i < 1001; i = 2 * i + 1) {
            mon[i] = true;
        }
        for (int j = 0; j <= n; j++) {
            if (mon[j]) {
                count++;
            }
        }
        return count;
    }
};
