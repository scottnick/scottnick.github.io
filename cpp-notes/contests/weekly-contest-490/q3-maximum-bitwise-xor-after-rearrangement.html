class Solution {
public:
    string maximumXor(string s, string t) {
        int dit[2] = {0};
        string ans = "";
        for (char a : t) {
            if (a == '0') {
                dit[0]++;
            } else {
                dit[1]++;
            }
        }
        for (char b : s) {
            if (b == '0') {
                if (dit[1] > 0) {
                    ans += '1';
                    dit[1]--;
                } else {
                    ans += '0';
                    dit[0]--;
                }
            } else {
                if (dit[0] > 0) {
                    ans += '1';
                    dit[0]--;
                } else {
                    ans += '0';
                    dit[1]--;
                }
            }
        }
        return ans;
    }
};
