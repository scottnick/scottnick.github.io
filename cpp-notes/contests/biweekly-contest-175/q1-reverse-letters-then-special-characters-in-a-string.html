class Solution {
public:
    string reverseByType(string s) {
        int n = s.size();
        vector<int> low;
        vector<int> spe;
        string ns = s;
        for (int i = 0; i < n; i++) {
            if (s[i] >= 'a' && s[i] <= 'z') {
                low.push_back(i);
            } else {
                spe.push_back(i);
            }
        }
        int nl = low.size();
        for (int j = 0; j < nl; j++) {
            ns[low[nl - 1 - j]] = s[low[j]];
        }
        int np = spe.size();
        for (int k = 0; k < np; k++) {
            ns[spe[np - 1 - k]] = s[spe[k]];
        }
        return ns;
    }
};
