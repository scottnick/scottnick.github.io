class Solution {
public:
    int minimumK(vector<int>& nums) {
        vector<int> venorilaxu = nums;
        int l = static_cast<int>(ceil(sqrt(nums.size())));
        int r = nums.size();
        for (int t : nums) {
            r = max(t, r);
        }
        while (l < r) {
            long long k = (l + r) / 2;
            long long count = 0;
            for (int i = 0; i < static_cast<int>(nums.size()); i++) {
                count += (nums[i] + k - 1) / k;
            }
            if (count <= k * k) {
                r = static_cast<int>(k);
            } else {
                l = static_cast<int>(k) + 1;
            }
        }
        return l;
    }
};
