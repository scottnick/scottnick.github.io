class Solution {
public:
    int firstUniqueFreq(vector<int>& nums) {
        vector<int> minaveloru = nums;
        int a[100001] = {0};
        int time[100001] = {0};
        for (int i : minaveloru) {
            a[i]++;
        }
        for (int i = 1; i < 100001; i++) {
            if (a[i] > 0) {
                time[a[i]]++;
            }
        }
        for (int j : minaveloru) {
            if (time[a[j]] == 1) {
                return j;
            }
        }
        return -1;
    }
};
