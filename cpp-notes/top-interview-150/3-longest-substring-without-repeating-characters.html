class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if (s.empty()) {
            return 0;
        }
        string sub = s.substr(0, 1);
        int ans = 0;
        for (int i = 1; i < s.size(); i++) {
            bool check = false;
            for (int j = 0; j < sub.size(); j++) {
                if (ans < sub.size()) {
                    ans = sub.size();
                }
                if (s[i] == sub[j]) {
                    sub = sub.substr(j + 1);
                    sub.push_back(s[i]);
                    check = true;
                    break;
                }
            }
            if (!check) {
                sub.push_back(s[i]);
            }
        }
        ans = max(ans, (int)sub.size());
        return ans;
    }
};
