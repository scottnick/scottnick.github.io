class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        bool same = true;
        for (int i = 0; i < ransomNote.size(); i++) {
            if (magazine.size() == 0) {
                same = false;
            }
            for (int j = 0; j < magazine.size(); j++) {
                if (ransomNote[i] == magazine[j]) {
                    magazine.erase(j, 1);
                    break;
                } else {
                    if (j == magazine.size() - 1) {
                        same = false;
                    }
                }
            }
        }
        return same;
    }
};
